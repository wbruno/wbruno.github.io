## Minha solução com apenas um único loop

<pre name="code" class="php">&lt;style type="text/css">
#grid td {
	width: 25px;
	height: 25px;
	text-align: center;
}
#grid td.diagonal {
	background: #000;
	color: #f00;
}
&lt;/style>
&lt;table id="grid">
&lt;tr>
&lt;?php
	$j = 1;
	$k = 1;
	for( $i=0; $i&lt;81; $i++ )
	{
		$ctrl = $i!=0 && $i%9==0;
		if( $ctrl ) echo '&lt;/tr>'.PHP_EOL.'&lt;tr>'.PHP_EOL;

		$class = $j==$k || $j+$k==10 ? ' class="diagonal"' : '';
		$text = $j==5 || $k==5 ? '+' : '-';

		echo "\t".'&lt;td'.$class.'>'.$text.'&lt;/td>'.PHP_EOL;

		$j = $j>=9 ? 1 : $j+1;
		$k = $j==1 ? $k+1 : $k;
	}
?>
&lt;/tr>
&lt;/table>
